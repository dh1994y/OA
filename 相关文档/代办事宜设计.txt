一、代办事宜表结构设计

每个用户一条记录
包括如下字段：
	id
	代办事宜
	最近心情
	创建时间
	创建人
	
	create table matter(
		id varchar(100) primary key comment 'id',
		matter varchar(255) comment '代办事宜',
		feel varchar(255) comment '最近心情',
		createDate date comment '创建日期',
		createUser varchar(100) comment '创建人id'
	);
